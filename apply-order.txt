CHAPTER 06 — CI/CD
========================================

    -- Enable the Pipelines console plugin (cluster-admin, one-time) --
    -- The operator installs the pipelines-console-plugin pod but doesn't
    -- always enable it in the Console config — without this, Developer
    -- perspective's Pipelines view stays empty even though `oc get pipeline`
    -- shows the Pipeline exists --
    oc get console.operator.openshift.io cluster -o jsonpath='{.spec.plugins}'
    -- if "pipelines-console-plugin" is missing from that list: --
    oc patch console.operator.openshift.io cluster --type=json \
      -p '[{"op": "add", "path": "/spec/plugins/-", "value": "pipelines-console-plugin"}]'

    -- Enable the GitOps console plugin too (same gap, same fix) --
    -- Without this, ArgoCD shows nothing in the OpenShift console even
    -- though `oc get application -n openshift-gitops` shows it Synced/Healthy --
    oc patch console.operator.openshift.io cluster --type=json \
      -p '[{"op": "add", "path": "/spec/plugins/-", "value": "gitops-plugin"}]'

    -- Create the GitHub webhook Secret imperatively --
    oc create secret generic github-webhook-secret \
      --from-literal=secret=$(openssl rand -hex 20)

    -- Create the git push credentials Secret and link it to financeflow-cicd --
    -- (needed by task-update-manifest.yaml's git-commit-push step — see
    --  secret-git-credentials.yaml comments for why no Task/workspace change
    --  is needed) --
    oc create secret generic git-credentials-cicd \
      --type=kubernetes.io/basic-auth \
      --from-literal=username=<your-github-username> \
      --from-literal=password=<your-github-PAT-with-repo-scope>
    oc annotate secret git-credentials-cicd tekton.dev/git-0=https://github.com
    oc secrets link financeflow-cicd git-credentials-cicd

68. chapters/07-observability/manifests/grafana/datasource.yaml
    (this CR alone leaves the datasource unauthenticated — see Step 7c2 below)

    -- Step 7c2: Patch the real bearer token into the datasource (one-time manual step) --
    -- Two operator-native mechanisms were tried and don't work for datasources
    -- created via the API (vs. file-based provisioning): $__env{BEARER_TOKEN}
    -- in secureJsonData, and spec.valuesFrom. Both confirmed dead ends live —
    -- see comments in datasource.yaml. PATCHing via Grafana's own API after
    -- the fact is what actually works: --
    GRAFANA_POD=$(oc get pod -n grafana -l app=financeflow-grafana -o jsonpath='{.items[0].metadata.name}')
    DS_UID=$(oc exec -n grafana "$GRAFANA_POD" -c grafana -- curl -s -u admin:financeflow \
      "http://localhost:3000/api/datasources" | python3 -c "import json,sys; print(json.load(sys.stdin)[0]['uid'])")
    TOKEN_RAW=$(oc get secret grafana-thanos-bearer-token -n grafana -o jsonpath='{.data.BEARER_TOKEN}' | base64 -d)
    oc exec -n grafana "$GRAFANA_POD" -c grafana -- curl -s -u admin:financeflow \
      -X PUT "http://localhost:3000/api/datasources/uid/$DS_UID" \
      -H "Content-Type: application/json" \
      -d "{\"name\":\"Prometheus\",\"type\":\"prometheus\",\"access\":\"proxy\",\"isDefault\":true,\"url\":\"https://thanos-querier.openshift-monitoring.svc.cluster.local:9091\",\"jsonData\":{\"httpMethod\":\"POST\",\"httpHeaderName1\":\"Authorization\",\"tlsSkipVerify\":true},\"secureJsonData\":{\"httpHeaderValue1\":\"$TOKEN_RAW\"}}"

    >> Verify: oc exec -n grafana "$GRAFANA_POD" -c grafana -- curl -s -u admin:financeflow \
               "http://localhost:3000/api/datasources/uid/$DS_UID/health"
    >>         (expect status: "OK", "Successfully queried the Prometheus API")
